DELIMITER $$

CREATE TRIGGER `before_insert_legal_debt_collector`
BEFORE INSERT ON `legal_debt_collector`
FOR EACH ROW
BEGIN
    DECLARE next_id INT;
    DECLARE formatted_code VARCHAR(150);

    -- Get the next auto-increment value from the same table
    SELECT IFNULL(MAX(id), 0) + 1 INTO next_id FROM legal_debt_collector;

    -- Format the code as 'TABL/DC/XXXX'
    SET formatted_code = CONCAT('TABL/DC/', LPAD(next_id, 4, '0'));

    -- Assign the formatted code to the new record
    SET NEW.code = formatted_code;
END$$

DELIMITER ;
