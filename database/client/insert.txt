CREATE TRIGGER `before_insert_legal_client`
BEFORE INSERT ON `legal_client`
FOR EACH ROW
BEGIN
    DECLARE next_id INT;
    DECLARE formatted_code VARCHAR(100);

    -- Get the next available ID safely
    SELECT IFNULL(MAX(id), 0) + 1 INTO next_id FROM legal_client;

    -- Format the code as 'TABL/CL/XXXX'
    SET formatted_code = CONCAT('TABL/CL/', LPAD(next_id, 4, '0'));

    -- Assign the formatted code to the new record
    SET NEW.code = formatted_code;
END$$

DELIMITER ;
